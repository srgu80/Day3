unning total
SELECT 
    product,
    sale_date,
    amount,
    SUM(amount) OVER (
        PARTITION BY product
        ORDER BY sale_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS running_total
FROM daily_sales
ORDER BY product, sale_date;

3 day average
SELECT
    product,
    sale_date,
    amount,
    AVG(amount) OVER (
        PARTITION BY product
        ORDER BY sale_date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg_3day
FROM daily_sales
ORDER BY product, sale_date;

LAG
SELECT
    product,
    sale_date,
    amount,
    LAG(amount, 1) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) AS prev_day_sales,
    amount - LAG(amount, 1) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) AS day_change
FROM daily_sales
ORDER BY product, sale_date;

Lead
SELECT
    product,
    sale_date,
    amount,
    LEAD(amount, 1) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) AS next_day_sales,
    LEAD(amount, 1) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) - amount AS next_day_change
FROM daily_sales
ORDER BY product, sale_date;

SELECT
    product,
    sale_date,
    amount,

    -- Running total
    SUM(amount) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) AS running_total,

    -- Moving average (3 days)
    AVG(amount) OVER (
        PARTITION BY product
        ORDER BY sale_date
        ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS moving_avg_3day,

    -- Previous day
    LAG(amount) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) AS prev_day,

    -- Next day
    LEAD(amount) OVER (
        PARTITION BY product
        ORDER BY sale_date
    ) AS next_day

FROM daily_sales
ORDER BY product, sale_date;

subquery - removal of duplicates
SELECT *
FROM (
    SELECT
        *,
        ROW_NUMBER() OVER (
            PARTITION BY customer_id, txn_date, amount
            ORDER BY txn_id
        ) AS rn
    FROM customer_transactions
) t
WHERE rn = 1
ORDER BY customer_id, txn_date;
